/**
 * @file maxOfThree.js
 * @description Finds the maximum among three numbers.
 * @version 1.0
 */

class MaxOfThree {
  /**
   * Creates a new MaxOfThree instance.
   */
  constructor() {}

  /**
   * Finds the maximum of three numbers.
   * @param {number} a - First number.
   * @param {number} b - Second number.
   * @param {number} c - Third number.
   * @returns {number} Maximum value.
   */
  findMax (a, b, c) {
    return Math.max(a, b, c);
  }

  /**
   * "Overload"-like variant: takes an array of exactly three numbers.
   * @param {number[]} values - Array of three numbers.
   * @returns {number} Maximum value.
   * @throws {RangeError} If the array length is not three.
   */
  findMaxFromArray (values) {
    if (!Array.isArray(values) || values.length !== 3) {
      throw new RangeError('values must be an array of exactly three numbers.');
    }
    const [a, b, c] = values;
    return this.findMax(a, b, c);
  }

  /**
   * "Overload"-like variant: accepts three numeric strings.
   * @param {string} a - First numeric string.
   * @param {string} b - Second numeric string.
   * @param {string} c - Third numeric string.
   * @returns {number} Maximum value.
   * @throws {TypeError} If any input is not a numeric string.
   */
  findMaxFromStrings (a, b, c) {
    const nums = [a, b, c].map(Number);
    if (nums.some((n) => Number.isNaN(n))) {
      throw new TypeError('All inputs must be numeric strings.');
    }
    return this.findMax(nums[0], nums[1], nums[2]);
  }
}

export default MaxOfThree;
