/**
 * @file e:\STM_Lab\SUBTRACTION\subtraction.js
 * @module Subtraction
 * @description Class providing subtraction operations with documented API following ES6 and JSDoc standards.
 * @author Jablay Noor Rahman (Omi)
 * @version 1.0
 */

class Subtraction {
  /**
   * Creates a new Subtraction object.
   * @param {number} [a=0] - First operand (optional).
   * @param {number} [b=0] - Second operand (optional).
   */
  constructor(a = 0, b = 0) {
    /**
     * First operand.
     * @type {number}
     */
    this.a = a;

    /**
     * Second operand.
     * @type {number}
     */
    this.b = b;

    /**
     * Last computed result.
     * @type {number}
     */
    this.result = 0;
  }

  /**
   * Subtract two numbers (generic).
   * @param {number} a - Minuend.
   * @param {number} b - Subtrahend.
   * @returns {number} Difference (a - b).
   * @throws {TypeError} If inputs are not finite numbers.
   * @example
   * const s = new Subtraction();
   * s.subtract(10, 3); // 7
   */
  subtract(a, b) {
    if (!Number.isFinite(a) || !Number.isFinite(b)) {
      throw new TypeError('subtract: both arguments must be finite numbers');
    }
    return a - b;
  }

  /**
   * Subtract two integers. Inputs are validated as integers.
   * @param {number} a - First integer.
   * @param {number} b - Second integer.
   * @returns {number} Difference of integers.
   * @throws {TypeError} If inputs are not integers.
   * @example
   * const s = new Subtraction();
   * s.subtractInt(10, 4); // 6
   */
  subtractInt(a, b) {
    if (!Number.isInteger(a) || !Number.isInteger(b)) {
      throw new TypeError('subtractInt: both arguments must be integers');
    }
    return a - b;
  }

  /**
   * Subtract two floating-point numbers.
   * @param {number} a - First number (float).
   * @param {number} b - Second number (float).
   * @returns {number} Difference of floating-point numbers.
   * @throws {TypeError} If inputs are not finite numbers.
   * @example
   * const s = new Subtraction();
   * s.subtractFloat(5.5, 2.2); // 3.3
   */
  subtractFloat(a, b) {
    if (!Number.isFinite(a) || !Number.isFinite(b)) {
      throw new TypeError('subtractFloat: both arguments must be finite numbers');
    }
    return a - b;
  }

  /**
   * Subtracts all numbers in an array sequentially: numbers[0] - numbers[1] - ...
   * @param {number[]} numbers - Array of numbers to subtract sequentially.
   * @returns {number} Result after sequential subtraction.
   * @throws {TypeError} If input is not a non-empty array of finite numbers.
   * @example
   * const s = new Subtraction();
   * s.subtractArray([20, 5, 3]); // 12
   */
  subtractArray(numbers) {
    if (!Array.isArray(numbers) || numbers.length === 0) {
      throw new TypeError('subtractArray: input must be a non-empty array');
    }
    if (!numbers.every(Number.isFinite)) {
      throw new TypeError('subtractArray: all elements must be finite numbers');
    }
    return numbers.slice(1).reduce((acc, curr) => acc - curr, numbers[0]);
  }

  /**
   * Performs subtraction using stored instance values (this.a - this.b).
   * Also updates this.result.
   * @returns {number} Result of this.a - this.b.
   * @example
   * const s = new Subtraction(15, 7);
   * s.subtractValues(); // 8
   */
  subtractValues() {
    this.result = this.subtract(this.a, this.b);
    return this.result;
  }

  /**
   * Display the result to console. If value omitted, display stored result.
   * @param {number} [value=this.result] - Value to display.
   * @returns {void}
   */
  display(value = this.result) {
    console.log(`Subtraction Result: ${value}`);
  }
}

export default Subtraction;
